import React, { useEffect } from "react";
// eslint-disable-next-line import/no-unresolved
import { inject, unmount } from "adminNext/dashboardInjector";

// Importing and mounting of the remote components should be done in
// the router pages, not in the main app.
const parentContainerid = "shadow-dom-container";

// todo: add a router here to add more pages?
const App = () => {
  // Mounts/unmounts the nextjs app into the "parent" container using remote "inject" function
  // https://github.com/module-federation/module-federation-examples/tree/master/different-react-versions-isolated
  useEffect(() => {
    inject(parentContainerid);
    return () => unmount(parentContainerid);
  }, []);

  return <div id={parentContainerid}></div>;
};

export default App;

import React from "react";
import { createRoot } from "react-dom/client";
import App from "./App";

/**
 * Mounts the React app into the root element with id 'dashboard-app'.
 */
export function setupReactApp() {
  const renderApp = () => {
    const rootElement = document.getElementById("dashboard-app");
    if (rootElement) {
      const root = createRoot(rootElement);
      root.render(
        <React.StrictMode>
          <App />
        </React.StrictMode>
      );
    } else {
      console.error("Could not find element with id 'dashboard-app'");
    }
  };

  if (document.readyState === "loading") {
    document.addEventListener("DOMContentLoaded", renderApp);
  } else {
    renderApp();
  }
}

import createErrorScreen from "./errorScreen";

// Check if we've already attempted a reload
const hasAttemptedReload = () => {
  const urlParams = new URLSearchParams(window.location.search);
  return urlParams.get("r") === "true";
};

// Perform hard refresh with reload parameter
const performHardRefresh = () => {
  const url = new URL(window.location.href);
  url.searchParams.set("r", "true");
  window.location.href = url.toString();
};

import("./bootstrap")
  .then(module => {
    console.log("Bootstrap module loaded [dashboard_app]");
    if (typeof module.setupReactApp === "function") {
      module.setupReactApp();
    } else {
      console.error("setupReactApp is not a function [dashboard_app]", module);
    }
  })
  .catch(error => {
    // Try hard refresh once, then show error screen
    if (!hasAttemptedReload()) {
      console.log("Module loading failed, attempting hard refresh...");
      performHardRefresh();
    } else {
      console.log("Hard refresh already attempted, showing error screen");
      createErrorScreen(error);
    }
  });

import React from "react";
import { createRoot } from "react-dom/client";
import { StyleSheetManager } from "styled-components";

import { createShadowContainer, deleteShadowContainer } from "./shadowDom";

// eslint-disable-next-line @typescript-eslint/no-explicit-any
const roots: Record<string, any> = {};

export const injectElement = (
  parentElementId: string,
  component: React.ReactNode
) => {
  const { appPlaceholder, shadowRoot } = createShadowContainer(parentElementId);
  if (appPlaceholder && shadowRoot) {
    const root = createRoot(appPlaceholder);
    roots[parentElementId] = root;
    root.render(
      // https://github.com/styled-components/styled-components/pull/1491
      <StyleSheetManager target={shadowRoot}>{component}</StyleSheetManager>
    );
  }
};

export const unmountElement = (parentElementId: string) => {
  deleteShadowContainer(parentElementId);
  const root = roots[parentElementId];
  if (root) {
    root.unmount();
    delete roots[parentElementId];
  }
};

  basePath: process.env.NEXT_PUBLIC_BASE_PATH,
  reactStrictMode: true,
  compiler: {
    styledComponents: true,
  },
  // This is required to support PostHog trailing slash API requests
  skipTrailingSlashRedirect: true,
  webpack(config, options) {
    if (!options.isServer) {
      config.plugins.push(
        new NextFederationPlugin({
          name: "adminNext",
          remotes: {},
          filename: "static/chunks/remoteEntry.js",
          exposes: {
            "./dashboardInjector":
              "./components/injectors/dashboardInjector.tsx",
          },
        })
      );
    }

    // Prevent dual styled-components instances with SC v6 exports field
    config.resolve.alias["styled-components"] = path.resolve(
      __dirname,
      "node_modules",
      "styled-components"
    );

    // Used to change public path which will be used to load the chunks, defaults to host of the page
    config.output.publicPath = `${process.env.NEXT_PUBLIC_BASE_PATH}/_next/`;
    return config;
  },
  async headers() {
    return [
      {
        source: "/_next/static/chunks/remoteEntry.js",
        headers: [
          {
            key: "Cache-Control",
            value:
              "no-store, no-cache, must-revalidate, proxy-revalidate, max-age=0",
          },
          {
            key: "Expires",
            value: "-1",
          },
        ],
      },
    ];
  },
  async rewrites() {
    const posthogRewrites = [
      // rewrites as reverse proxies - https://posthog.com/docs/advanced/proxy/nextjs
      {
        source: "/api/is/static/:path*",
        destination: "https://eu-assets.i.posthog.com/static/:path*",
      },
      {
        source: "/api/is/:path*",
        destination: "https://eu.i.posthog.com/:path*",
      },
    ];

    const apiRewrites = [
      {
        source: process.env.NEXT_PUBLIC_API_ENDPOINT,
        destination: process.env.API_HOST,
        basePath: false,
      },
    ];

    // In production: only PostHog rewrites
    if (process.env.NODE_ENV === "production") {
      return posthogRewrites;
    }

    // In development: both API and PostHog rewrites work
    return [...apiRewrites, ...posthogRewrites];
  },
};
